def gene_list_coords(gene_list):
    append_list = []
    for gene in gene_list:
        coord = gi.gene_coord(gene, assembly='hg38')
        if coord:
            chrom, start, end, strand = gi.gene_coord(gene, assembly='hg38')[gene]
            append_list.append((chrom, (start + end)/2, gene))
        else:
            print('skipped', gene)
    return append_list
